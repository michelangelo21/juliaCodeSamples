173 function sincos(x::T) where T<:Union{Float32, Float64}
174 if abs(x) < T(pi)/4
175 if x == zero(T)
176 return x, one(T)
177 end
178 return sincos_kernel(x)
179 elseif isnan(x)
180 return T(NaN), T(NaN)
181 elseif isinf(x)
182 sincos_domain_error(x)
183 end
184 n, y = rem_pio2_kernel(x)
185 n = n&3
186 # calculate both kernels at the reduced y...
187 si, co = sincos_kernel(y)
188 # ... and use the same selection scheme as above: (sin, cos, -sin, -cos) for
189 # for sin and (cos, -sin, -cos, sin) for cos
190 if n == 0
191 return si, co
192 elseif n == 1
193 return co, -si
194 elseif n == 2
195 return -si, -co
196 else
197 return -co, si
198 end
199 end

236 @inline function tan_kernel(y::DoubleFloat64, k)
237 # kernel tan function on ~[-pi/4, pi/4] (except on -0)
238 # Input y is assumed to be bounded by ~pi/4 in magnitude.
239 # Input k indicates whether tan (if k = 1) or -1/tan (if k = -1) is returned.
240
241 # Algorithm
242 # 1. Since tan(-y) = -tan(y), we need only to consider positive y.
243 # 2. Callers must return tan(-0) = -0 without calling here since our
244 # odd polynomial is not evaluated in a way that preserves -0.
245 # Callers may do the optimization tan(y) ~ y for tiny y.
246 # 3. tan(y) is approximated by a odd polynomial of degree 27 on
247 # [0,0.67434]
248 # 3 27
249 # tan(y) ~ y + T1*y + ... + T13*y ≡ P(y)
250 # where
251 #
252 # |tan(y) 2 4 26 | -59.2
253 # (tan(y)-P(y))/y = |----- - (1+T1*y +T2*y +.... +T13*y )| <= 2
254 # | y |
255 #
256 # Note: tan(y+z) = tan(y) + tan'(y)*z
257 # ~ tan(y) + (1+y*y)*z
258 # Therefore, for better accuracz in computing tan(y+z), let
259 # 3 2 2 2 2
260 # r = y *(T2+y *(T3+y *(...+y *(T12+y *T13))))
261 # then
262 # 3 2
263 # tan(y+z) = y + (T1*y + (y *(r+z)+z))
264 #
265 # 4. For y in [0.67434,pi/4], let z = pi/4 - y, then
266 # tan(y) = tan(pi/4-z) = (1-tan(z))/(1+tan(z))
267 # = 1 - 2*(tan(z) - (tan(z)^2)/(1+tan(z)))
268
269 yhi = y.hi
270 ylo = y.lo
271
272 if abs(yhi) >= 0.6744
273 if yhi < 0.0
274 yhi = -yhi
275 ylo = -ylo
276 end
277 # Then, accurately reduce y as "pio4hi"-yhi+"pio4lo"-ylo
278 yhi = (pi/4 - yhi) + (3.06161699786838301793e-17 - ylo)
279 # yhi is guaranteed to be exact, so ylo is identically zero
280 ylo = 0.0
281 end
282 y² = yhi * yhi
283 y⁴ = y² * y²
284
285 # Break P(y)-T1*y³ = y^5*(T[2]+y^2*T[3]+...) into y⁵*r + y⁵*v where
286 # r = T[2]+y^4*T[4]+...+y^20*T[12])
287 # v = (y^2*(T[3]+y^4*T[5]+...+y^22*[T13]))
288 r = @horner(y⁴,
289 1.33333333333201242699e-01, # T2
290 2.18694882948595424599e-02, # T4
291 3.59207910759131235356e-03, # T6
292 5.88041240820264096874e-04, # T8
293 7.81794442939557092300e-05, # T10
294 -1.85586374855275456654e-05) # T12
295 v = y² * @horner(y⁴,
296 5.39682539762260521377e-02, # T3
297 8.86323982359930005737e-03, # T5
298 1.45620945432529025516e-03, # T7
299 2.46463134818469906812e-04, # T9
300 7.14072491382608190305e-05, # T11
301 2.59073051863633712884e-05) # T13
302 # Precompute y³
303 y³ = y² * yhi
304 # Calculate P(y)-y-T1*y³ = y⁵*r + y⁵*v = y²(y³*(r+v))
305 r = ylo + y² * (y³ * (r + v) + ylo)
306 # Calculate P(y)-y = r+T1*y³
307 r += 3.33333333333334091986e-01*y³
308 # Calculate w = r+y = P(y)
309 Px = yhi + r
310 if abs(y.hi) >= 0.6744
311 # If the original y was above the threshold, then we calculate
312 # tan(y) = 1 - 2*(tan(y) - (tan(y)^2)/(1+tan(y)))
313 # ≈ 1 - 2*(P(z) - (P(z)^2)/(1+P(z)))
314 # where z = y-π/4.
315 return (signbit(y.hi) ? -1.0 : 1.0)*(k - 2*(yhi-(Px^2/(k+Px)-r)))
316 end
317 if k == 1
318 # Else, we simply return w = P(y) if k == 1 (integer multiple from argument
319 # reduction was even)...
320 return Px
321 else
322 # ...or tan(y) ≈ -1.0/(y+r) if !(k == 1) (integer multiple from argument
323 # reduction was odd). If 2ulp error is allowed, simply return the frac-
324 # tion directly. Instead, we calculate it accurately.
325
326 # Px0 is w with zeroed out low word
327 Px0 = reinterpret(Float64, (reinterpret(UInt64, Px) >> 32) << 32)
328 v = r - (Px0 - yhi) # Px0+v = r+y
329 t = a = -1.0 / Px
330 # zero out low word of t
331 t = reinterpret(Float64, (reinterpret(UInt64, t) >> 32) << 32)
332 s = 1.0 + t * Px0
333 return t + a * (s + t * v)
334 end
335 end

393 @inline function asin_kernel(t::Float64, x::Float64)
394 # we use that for 1/2 <= x < 1 we have
395 # asin(x) = pi/2-2*asin(sqrt((1-x)/2))
396 # Let y = (1-x), z = y/2, s := sqrt(z), and pio2_hi+pio2_lo=pi/2;
397 # then for x>0.98
398 # asin(x) = pi/2 - 2*(s+s*z*R(z))
399 # = pio2_hi - (2*(s+s*z*R(z)) - pio2_lo)
400 # For x<=0.98, let pio4_hi = pio2_hi/2, then
401 # f = hi part of s;
402 # c = sqrt(z) - f = (z-f*f)/(s+f) ...f+c=sqrt(z)
403 # and
404 # asin(x) = pi/2 - 2*(s+s*z*R(z))
405 # = pio4_hi+(pio4-2s)-(2s*z*R(z)-pio2_lo)
406 # = pio4_hi+(pio4-2f)-(2s*z*R(z)-(pio2_lo+2c))
407 pio2_lo = 6.12323399573676603587e-17
408 s = sqrt_llvm(t)
409 tRt = arc_tRt(t)
410 if abs(x) >= 0.975 # |x| > 0.975
411 return flipsign(pi/2 - (2.0*(s + s*tRt) - pio2_lo), x)
412 else
413 s0 = reinterpret(Float64, (reinterpret(UInt64, s) >> 32) << 32)
414 c = (t - s0*s0)/(s + s0)
415 p = 2.0*s*tRt - (pio2_lo - 2.0*c)
416 q = pi/4 - 2.0*s0
417 return flipsign(pi/4 - (p-q), x)
418 end
419 end

426 @noinline asin_domain_error(x) = throw(DomainError(x, "asin(x) is not defined for |x|>1."))
427 function asin(x::T) where T<:Union{Float32, Float64}
428 # Since asin(x) = x + x^3/6 + x^5*3/40 + x^7*15/336 + ...
429 # we approximate asin(x) on [0,0.5] by
430 # asin(x) = x + x*x^2*R(x^2)
431 # where
432 # R(x^2) is a rational approximation of (asin(x)-x)/x^3
433 # and its remez error is bounded by
434 # |(asin(x)-x)/x^3 - R(x^2)| < 2^(-58.75)
435 absx = abs(x)
436 if absx >= T(1.0) # |x|>= 1
437 if absx == T(1.0)
438 return flipsign(T(pi)/2, x)
439 end
440 asin_domain_error(x)
441 elseif absx < T(1.0)/2
442 # if |x| sufficiently small, |x| is a good approximation
443 if absx < ASIN_X_MIN_THRESHOLD(T)
444 return x
445 end
446 return muladd(x, arc_tRt(x*x), x)
447 end
448 # else 1/2 <= |x| < 1
449 t = (T(1.0) - absx)/2
450 return asin_kernel(t, x)
451 end

502 atan(x::Real) = atan(float(x))
503 function atan(x::T) where T<:Union{Float32, Float64}
504 # Method
505 # 1. Reduce x to positive by atan(x) = -atan(-x).
506 # 2. According to the integer k=4t+0.25 chopped, t=x, the argument
507 # is further reduced to one of the following intervals and the
508 # arctangent of t is evaluated by the corresponding formula:
509 #
510 # [0,7/16] atan(x) = t-t^3*(a1+t^2*(a2+...(a10+t^2*a11)...)
511 # [7/16,11/16] atan(x) = atan(1/2) + atan( (t-0.5)/(1+t/2) )
512 # [11/16.19/16] atan(x) = atan( 1 ) + atan( (t-1)/(1+t) )
513 # [19/16,39/16] atan(x) = atan(3/2) + atan( (t-1.5)/(1+1.5t) )
514 # [39/16,INF] atan(x) = atan(INF) + atan( -1/t )
515 #
516 # If isnan(x) is true, then the nan value will eventually be passed to
517 # atan_pq(x) and return the appropriate nan value.
518
519 absx = abs(x)
520 if absx >= ATAN_LARGE_X(T)
521 return copysign(T(1.5707963267948966), x)
522 end
523 if absx < T(7/16)
524 # no reduction needed
525 if absx < ATAN_SMALL_X(T)
526 return x
527 end
528 p, q = atan_pq(x)
529 return x - x*(p + q)
530 end
531 xsign = sign(x)
532 if absx < T(19/16) # 7/16 <= |x| < 19/16
533 if absx < T(11/16) # 7/16 <= |x| <11/16
534 hi = ATAN_1_O_2_HI(T)
535 lo = ATAN_1_O_2_LO(T)
536 x = (T(2.0)*absx - T(1.0))/(T(2.0) + absx)
537 else # 11/16 <= |x| < 19/16
538 hi = ATAN_2_O_2_HI(T)
539 lo = ATAN_2_O_2_LO(T)
540 x = (absx - T(1.0))/(absx + T(1.0))
541 end
542 else
543 if absx < T(39/16) # 19/16 <= |x| < 39/16
544 hi = ATAN_3_O_2_HI(T)
545 lo = ATAN_3_O_2_LO(T)
546 x = (absx - T(1.5))/(T(1.0) + T(1.5)*absx)
547 else # 39/16 <= |x| < upper threshold (2.0^66 or 2.0f0^26)
548 hi = ATAN_INF_HI(T)
549 lo = ATAN_INF_LO(T)
550 x = -T(1.0)/absx
551 end
552 end
553 # end of argument reduction
554 p, q = atan_pq(x)
555 z = hi - ((x*(p + q) - lo) - x)
556 copysign(z, xsign)
557 end

567 function atan(y::T, x::T) where T<:Union{Float32, Float64}
568 # Method :
569 # M1) Reduce y to positive by atan2(y,x)=-atan2(-y,x).
570 # M2) Reduce x to positive by (if x and y are unexceptional):
571 # ARG (x+iy) = arctan(y/x) ... if x > 0,
572 # ARG (x+iy) = pi - arctan[y/(-x)] ... if x < 0,
573 #
574 # Special cases:
575 #
576 # S1) ATAN2((anything), NaN ) is NaN;
577 # S2) ATAN2(NAN , (anything) ) is NaN;
578 # S3) ATAN2(+-0, +(anything but NaN)) is +-0 ;
579 # S4) ATAN2(+-0, -(anything but NaN)) is +-pi ;
580 # S5) ATAN2(+-(anything but 0 and NaN), 0) is +-pi/2;
581 # S6) ATAN2(+-(anything but INF and NaN), +INF) is +-0 ;
582 # S7) ATAN2(+-(anything but INF and NaN), -INF) is +-pi;
583 # S8) ATAN2(+-INF,+INF ) is +-pi/4 ;
584 # S9) ATAN2(+-INF,-INF ) is +-3pi/4;
585 # S10) ATAN2(+-INF, (anything but,0,NaN, and INF)) is +-pi/2;
586 if isnan(x) || isnan(y) # S1 or S2
587 return T(NaN)
588 end
589
590 if x == T(1.0) # then y/x = y and x > 0, see M2
591 return atan(y)
592 end
593 # generate an m ∈ {0, 1, 2, 3} to branch off of
594 m = 2*signbit(x) + 1*signbit(y)
595
596 if iszero(y)
597 if m == 0 || m == 1
598 return y # atan(+-0, +anything) = +-0
599 elseif m == 2
600 return T(pi) # atan(+0, -anything) = pi
601 elseif m == 3
602 return -T(pi) # atan(-0, -anything) =-pi
603 end
604 elseif iszero(x)
605 return flipsign(T(pi)/2, y)
606 end
607
608 if isinf(x)
609 if isinf(y)
610 if m == 0
611 return T(pi)/4 # atan(+Inf), +Inf))
612 elseif m == 1
613 return -T(pi)/4 # atan(-Inf), +Inf))
614 elseif m == 2
615 return 3*T(pi)/4 # atan(+Inf, -Inf)
616 elseif m == 3
617 return -3*T(pi)/4 # atan(-Inf,-Inf)
618 end
619 else
620 if m == 0
621 return zero(T) # atan(+...,+Inf) */
622 elseif m == 1
623 return -zero(T) # atan(-...,+Inf) */
624 elseif m == 2
625 return T(pi) # atan(+...,-Inf) */
626 elseif m == 3
627 return -T(pi) # atan(-...,-Inf) */
628 end
629 end
630 end
631
632 # x wasn't Inf, but y is
633 isinf(y) && return copysign(T(pi)/2, y)
634
635 ypw = poshighword(y)
636 xpw = poshighword(x)
637 # compute y/x for Float32
638 k = reinterpret(Int32, ypw-xpw)>>ATAN2_RATIO_BIT_SHIFT(T)
639
640 if k > ATAN2_RATIO_THRESHOLD(T) # |y/x| > threshold
641 z=T(pi)/2+T(0.5)*ATAN2_PI_LO(T)
642 m&=1;
643 elseif x<0 && k < -ATAN2_RATIO_THRESHOLD(T) # 0 > |y|/x > threshold
644 z = zero(T)
645 else #safe to do y/x
646 z = atan(abs(y/x))
647 end
648
649 if m == 0
650 return z # atan(+,+)
651 elseif m == 1
652 return -z # atan(-,+)
653 elseif m == 2
654 return T(pi)-(z-ATAN2_PI_LO(T)) # atan(+,-)
655 else # default case m == 3
656 return (z-ATAN2_PI_LO(T))-T(pi) # atan(-,-)
657 end
658 end

660 ACOS_X_MIN_THRESHOLD(::Type{Float32}) = 2.0f0^-26
661 ACOS_X_MIN_THRESHOLD(::Type{Float64}) = 2.0^-57
662 PIO2_HI(::Type{Float32}) = 1.5707962513f+00
663 PIO2_LO(::Type{Float32}) = 7.5497894159f-08
664 PIO2_HI(::Type{Float64}) = 1.57079632679489655800e+00
665 PIO2_LO(::Type{Float64}) = 6.12323399573676603587e-17
666 ACOS_PI(::Type{Float32}) = 3.1415925026f+00
667 ACOS_PI(::Type{Float64}) = 3.14159265358979311600e+00
668 @inline ACOS_CORRECT_LOWWORD(::Type{Float32}, x) = reinterpret(Float32, (reinterpret(UInt32, x) & 0xfffff000))
669 @inline ACOS_CORRECT_LOWWORD(::Type{Float64}, x) = reinterpret(Float64, (reinterpret(UInt64, x) >> 32) << 32)

671 @noinline acos_domain_error(x) = throw(DomainError(x, "acos(x) not defined for |x| > 1"))
672 function acos(x::T) where T <: Union{Float32, Float64}
673 # Method :
674 # acos(x) = pi/2 - asin(x)
675 # acos(-x) = pi/2 + asin(x)
676 # As a result, we use the same rational approximation (arc_tRt) as in asin.
677 # See the comments in asin for more information about this approximation.
678 # 1) For |x| <= 0.5
679 # acos(x) = pi/2 - (x + x*x^2*R(x^2))
680 # 2) For x < -0.5
681 # acos(x) = pi - 2asin(sqrt((1 - |x|)/2))
682 # = pi - 0.5*(s+s*z*R(z))
683 # where z=(1-|x|)/2, s=sqrt(z)
684 # 3) For x > 0.5
685 # acos(x) = pi/2 - (pi/2 - 2asin(sqrt((1 - x)/2)))
686 # = 2asin(sqrt((1 - x)/2))
687 # = 2s + 2s*z*R(z) ...z=(1 - x)/2, s=sqrt(z)
688 # = 2f + (2c + 2s*z*R(z))
689 # where f=hi part of s, and c = (z - f*f)/(s + f) is the correction term
690 # for f so that f + c ~ sqrt(z).
691
692 # Special cases:
693 # 4) if x is NaN, return x itself;
694 # 5) if |x|>1 throw warning.
695
696 absx = abs(x)
697 if absx >= T(1.0)
698 # acos(-1) = π, acos(1) = 0
699 absx == T(1.0) && return x > T(0.0) ? T(0.0) : T(pi)
700 # acos(x) is not defined for |x| > 1
701 acos_domain_error(x) # see 5) above
702 elseif absx < T(1.0)/2 # see 1) above
703 # if |x| sufficiently small, acos(x) ≈ pi/2
704 absx < ACOS_X_MIN_THRESHOLD(T) && return T(pi)/2
705 # if |x| < 0.5 we have acos(x) = pi/2 - (x + x*x^2*R(x^2))
706 return PIO2_HI(T) - (x - (PIO2_LO(T) - x*arc_tRt(x*x)))
707 end
708 z = (T(1.0) - absx)*T(0.5)
709 zRz = arc_tRt(z)
710 s = sqrt_llvm(z)
711 if x < T(0.0) # see 2) above
712 return ACOS_PI(T) - T(2.0)*(s + (zRz*s - PIO2_LO(T)))
713 else # see 3) above
714 # if x > 0.5 we have
715 # acos(x) = pi/2 - (pi/2 - 2asin(sqrt((1-x)/2)))
716 # = 2asin(sqrt((1-x)/2))
717 # = 2s + 2s*z*R(z) ...z=(1-x)/2, s=sqrt(z)
718 # = 2f + (2c + 2s*z*R(z))
719 # where f=hi part of s, and c = (z-f*f)/(s+f) is the correction term
720 # for f so that f+c ~ sqrt(z).
721 df = ACOS_CORRECT_LOWWORD(T, s)
722 c = (z - df*df)/(s + df)
723 return T(2.0)*(df + (zRz*s + c))
724 end
725 end

773 copysign(cos_kernel(y),rx)
774 else
775 y = mulpi_ext(copysign(one(T),rx) - rx)
776 sin_kernel(y)
777 end
