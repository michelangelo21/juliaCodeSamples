36 gc_total_bytes(gc_num::GC_Num) =
37 gc_num.allocd + gc_num.deferred_alloc + gc_num.total_allocd

39 function GC_Diff(new::GC_Num, old::GC_Num)
40 # logic from `src/gc.c:jl_gc_total_bytes`
41 old_allocd = gc_total_bytes(old)
42 new_allocd = gc_total_bytes(new)
43 return GC_Diff(new_allocd - old_allocd,
44 new.malloc - old.malloc,
45 new.realloc - old.realloc,
46 new.poolalloc - old.poolalloc,
47 new.bigalloc - old.bigalloc,
48 new.freecall - old.freecall,
49 new.total_time - old.total_time,
50 new.pause - old.pause,
51 new.full_sweep - old.full_sweep)
52 end

54 function gc_alloc_count(diff::GC_Diff)
55 diff.malloc + diff.realloc + diff.poolalloc + diff.bigalloc
56 end

77 const _cnt_units = ["", " k", " M", " G", " T", " P"]
78 function prettyprint_getunits(value, numunits, factor)
79 if value == 0 || value == 1
80 return (value, 1)
81 end
82 unit = ceil(Int, log(value) / log(factor))
83 unit = min(numunits, unit)
84 number = value/factor^(unit-1)
85 return number, unit
86 end

96 function format_bytes(bytes) # also used by InteractiveUtils
97 bytes, mb = prettyprint_getunits(bytes, length(_mem_units), Int64(1024))
98 if mb == 1
99 return string(Int(bytes), " ", _mem_units[mb], bytes==1 ? "" : "s")
100 else
101 return string(Ryu.writefixed(Float64(bytes), 3), " ", _mem_units[mb])
102 end
103 end

105 function time_print(elapsedtime, bytes=0, gctime=0, allocs=0)
106 timestr = Ryu.writefixed(Float64(elapsedtime/1e9), 6)
107 length(timestr) < 10 && print(" "^(10 - length(timestr)))
108 print(timestr, " seconds")
109 if bytes != 0 || allocs != 0
110 allocs, ma = prettyprint_getunits(allocs, length(_cnt_units), Int64(1000))
111 if ma == 1
112 print(" (", Int(allocs), _cnt_units[ma], allocs==1 ? " allocation: " : " allocations: ")
113 else
114 print(" (", Ryu.writefixed(Float64(allocs), 2), _cnt_units[ma], " allocations: ")
115 end
116 print(format_bytes(bytes))
117 end
118 if gctime > 0
119 print(", ", Ryu.writefixed(Float64(100*gctime/elapsedtime), 2), "% gc time")
120 end
121 if bytes != 0 || allocs != 0
122 print(")")
123 end
124 nothing
125 end

127 function timev_print(elapsedtime, diff::GC_Diff)
128 allocs = gc_alloc_count(diff)
129 time_print(elapsedtime, diff.allocd, diff.total_time, allocs)
130 print("\nelapsed time (ns): $elapsedtime\n")
131 padded_nonzero_print(diff.total_time, "gc time (ns)")
132 padded_nonzero_print(diff.allocd, "bytes allocated")
133 padded_nonzero_print(diff.poolalloc, "pool allocs")
134 padded_nonzero_print(diff.bigalloc, "non-pool GC allocs")
135 padded_nonzero_print(diff.malloc, "malloc() calls")
136 padded_nonzero_print(diff.realloc, "realloc() calls")
137 padded_nonzero_print(diff.freecall, "free() calls")
138 padded_nonzero_print(diff.pause, "GC pauses")
139 padded_nonzero_print(diff.full_sweep, "full collections")
140 end

142 """
143 @time
144
145 A macro to execute an expression, printing the time it took to execute, the number of
146 allocations, and the total number of bytes its execution caused to be allocated, before
147 returning the value of the expression.
148
149 See also [`@timev`](@ref), [`@timed`](@ref), [`@elapsed`](@ref), and
150 [`@allocated`](@ref).
151
152 !!! note
153 For more serious benchmarking, consider the `@btime` macro from the BenchmarkTools.jl
154 package which among other things evaluates the function multiple times in order to
155 reduce noise.
156
157 ```julia-repl
158 julia> @time rand(10^6);
159 0.001525 seconds (7 allocations: 7.630 MiB)
160
161 julia> @time begin
162 sleep(0.3)
163 1+1
164 end
165 0.301395 seconds (8 allocations: 336 bytes)
166 2
167 ```
168 """

172 local stats = gc_num()
173 local elapsedtime = time_ns()
174 621 (0.90%) 67714 (98.54%)
67714 (98.54%) samples spent in macro expansion
621 (100.00%) (ex.), 67714 (100.00%) (incl.) when called from macro expansion line 93
67093 (99.08%) samples spent calling nlc
local val = $(esc(ex))
175 elapsedtime = time_ns() - elapsedtime
176 local diff = GC_Diff(gc_num(), stats)
177 time_print(elapsedtime, diff.allocd, diff.total_time,
178 gc_alloc_count(diff))
179 println()
180 val
181 end
182 end

187 This is a verbose version of the `@time` macro. It first prints the same information as

305 macro timed(ex)

308 local stats = gc_num()
309 local elapsedtime = time_ns()

311 elapsedtime = time_ns() - elapsedtime
