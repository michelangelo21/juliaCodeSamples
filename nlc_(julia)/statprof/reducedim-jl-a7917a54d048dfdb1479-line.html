129 @eval function reducedim_init(f, op::typeof($f1), A::AbstractArray, region)
130 # First compute the reduce indices. This will throw an ArgumentError
131 # if any region is invalid
132 ri = reduced_indices(A, region)
133
134 # Next, throw if reduction is over a region with length zero
135 any(i -> isempty(axes(A, i)), region) && _empty_reduce_error()
136
137 # Make a view of the first slice of the region
138 A1 = view(A, ri...)
139
140 if isempty(A1)
141 # If the slice is empty just return non-view version as the initial array
142 return copy(A1)
143 else
144 # otherwise use the min/max of the first slice as initial value
145 v0 = mapreduce(f, $f2, A1)
146
147 # but NaNs need to be avoided as initial values
148 v0 = v0 != v0 ? typeof(v0)($initval) : v0
149
150 T = _realtype(f, promote_union(eltype(A)))
151 Tr = v0 isa T ? T : typeof(v0)
152 return reducedim_initarray(A, region, v0, Tr)
153 end
154 end

188 function check_reducedims(R, A)
189 # Check whether R has compatible dimensions w.r.t. A for reduction
190 #
191 # It returns an integer value (useful for choosing implementation)
192 # - If it reduces only along leading dimensions, e.g. sum(A, dims=1) or sum(A, dims=(1,2)),
193 # it returns the length of the leading slice. For the two examples above,
194 # it will be size(A, 1) or size(A, 1) * size(A, 2).
195 # - Otherwise, e.g. sum(A, dims=2) or sum(A, dims=(1,3)), it returns 0.
196 #
197 ndims(R) <= ndims(A) || throw(DimensionMismatch("cannot reduce $(ndims(A))-dimensional array to $(ndims(R)) dimensions"))
198 lsiz = 1
199 had_nonreduc = false
200 for i = 1:ndims(A)
201 Ri, Ai = axes(R, i), axes(A, i)
202 sRi, sAi = length(Ri), length(Ai)
203 if sRi == 1
204 if sAi > 1
205 if had_nonreduc
206 lsiz = 0 # to reduce along i, but some previous dimensions were non-reducing
207 else
208 lsiz *= sAi # if lsiz was set to zero, it will stay to be zero
209 end
210 end
211 else
212 Ri == Ai || throw(DimensionMismatch("reduction on array with indices $(axes(A)) with output with indices $(axes(R))"))
213 had_nonreduc = true
214 end
215 end
216 return lsiz
217 end

239 function _mapreducedim!(f, op, R::AbstractArray, A::AbstractArrayOrBroadcasted)
240 lsiz = check_reducedims(R,A)
241 isempty(A) && return R
242
243 if has_fast_linear_indexing(A) && lsiz > 16
244 # use mapreduce_impl, which is probably better tuned to achieve higher performance
245 nslices = div(length(A), lsiz)
246 ibase = first(LinearIndices(A))-1
247 for i = 1:nslices
248 @inbounds R[i] = op(R[i], mapreduce_impl(f, op, A, ibase+1, ibase+lsiz))
249 ibase += lsiz
250 end
251 return R
252 end
253 indsAt, indsRt = safe_tail(axes(A)), safe_tail(axes(R)) # handle d=1 manually
254 keep, Idefault = Broadcast.shapeindexer(indsRt)
255 if reducedim1(R, A)
256 # keep the accumulator as a local variable when reducing along the first dimension
257 i1 = first(axes1(R))
258 @inbounds for IA in CartesianIndices(indsAt)
259 IR = Broadcast.newindex(IA, keep, Idefault)
260 r = R[i1,IR]
261 @simd for i in axes(A, 1)
262 r = op(r, f(A[i, IA]))
263 end
264 R[i1,IR] = r
265 end
266 else
267 @inbounds for IA in CartesianIndices(indsAt)
268 IR = Broadcast.newindex(IA, keep, Idefault)
269 @simd for i in axes(A, 1)
270 R[i,IR] = op(R[i,IR], f(A[i,IA]))
271 end
272 end
273 end
274 return R
275 end

754 # The initial values of Rval are not used if the corresponding indices in Rind are 0.
755 #
756 function findminmax!(f, Rval, Rind, A::AbstractArray{T,N}) where {T,N}
757 (isempty(Rval) || isempty(A)) && return Rval, Rind
758 lsiz = check_reducedims(Rval, A)
759 for i = 1:N
760 axes(Rval, i) == axes(Rind, i) || throw(DimensionMismatch("Find-reduction: outputs must have the same indices"))
761 end
762 # If we're reducing along dimension 1, for efficiency we can make use of a temporary.
763 # Otherwise, keep the result in Rval/Rind so that we traverse A in storage order.
764 indsAt, indsRt = safe_tail(axes(A)), safe_tail(axes(Rval))
765 keep, Idefault = Broadcast.shapeindexer(indsRt)
766 ks = keys(A)
767 y = iterate(ks)
768 zi = zero(eltype(ks))
769 if reducedim1(Rval, A)
770 i1 = first(axes1(Rval))
771 @inbounds for IA in CartesianIndices(indsAt)
772 IR = Broadcast.newindex(IA, keep, Idefault)
773 tmpRv = Rval[i1,IR]
774 tmpRi = Rind[i1,IR]
775 for i in axes(A,1)
776 k, kss = y::Tuple
777 tmpAv = A[i,IA]
778 if tmpRi == zi || (tmpRv == tmpRv && (tmpAv != tmpAv || f(tmpAv, tmpRv)))
779 tmpRv = tmpAv
780 tmpRi = k
781 end
782 y = iterate(ks, kss)
783 end
784 Rval[i1,IR] = tmpRv
785 Rind[i1,IR] = tmpRi
786 end
787 else
788 @inbounds for IA in CartesianIndices(indsAt)
789 IR = Broadcast.newindex(IA, keep, Idefault)
790 for i in axes(A, 1)
791 k, kss = y::Tuple
792 tmpAv = A[i,IA]
793 tmpRv = Rval[i,IR]
794 tmpRi = Rind[i,IR]
795 if tmpRi == zi || (tmpRv == tmpRv && (tmpAv != tmpAv || f(tmpAv, tmpRv)))
796 Rval[i,IR] = tmpAv
797 Rind[i,IR] = k
798 end
799 y = iterate(ks, kss)
800 end
801 end
802 end
